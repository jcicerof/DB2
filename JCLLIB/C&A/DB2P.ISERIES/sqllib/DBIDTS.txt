--
-- TRANSLATE DAS TABELAS
--
--
SELECT S.NAME,STRIP(CHAR(S.DBID))||','||STRIP(CHAR(D.DBID)), 1
  FROM SYSIBM.SYSTABLESPACE S
     , SYSIBM.SYSTABLESPACE D
 WHERE S.NAME = 'EQ001T'     AND
       S.DBNAME <> 'AISERIES'  AND
       D.NAME = S.NAME       AND
       D.DBNAME = 'AISERIES'
UNION ALL
SELECT S.NAME,STRIP(CHAR(S.PSID))||','||STRIP(CHAR(D.PSID)), 2
  FROM SYSIBM.SYSTABLESPACE S
     , SYSIBM.SYSTABLESPACE D
 WHERE S.NAME = 'EQ001T'     AND
       S.DBNAME = 'EQ0BP'    AND
       S.DBNAME <> 'AISERIES'  AND
       D.NAME = S.NAME       AND
       D.DBNAME = 'AISERIES'
UNION ALL
SELECT S.TSNAME,STRIP(CHAR(S.OBID))||','||STRIP(CHAR(D.OBID)), 3
  FROM SYSIBM.SYSTABLES S
     , SYSIBM.SYSTABLES D
 WHERE S.TSNAME = 'EQ001T'  AND
       S.CREATOR = 'PROD'   AND
       D.TSNAME = S.TSNAME  AND
       D.CREATOR = 'SUPPROD'
ORDER BY 1, 3
;
