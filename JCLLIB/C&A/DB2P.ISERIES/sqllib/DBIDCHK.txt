-- INDICES
SELECT INDEXSPACE,NAME,DBID, ISOBID, OBID FROM SYSIBM.SYSINDEXES
WHERE TBNAME LIKE 'EQ035%'
AND CREATOR='PROD';
SELECT INDEXSPACE,NAME,DBID, ISOBID, OBID FROM SYSIBM.SYSINDEXES
WHERE TBNAME LIKE 'EQ035%'
AND CREATOR='SUPPROD';
-- TABLESPACES
SELECT PARTITION, PQTY, SQTY FROM SYSIBM.SYSTABLEPART
 WHERE TSNAME LIKE 'EQ035%'        -- 'TABLESPACE_NAME'
   AND DBNAME='EQ0BP';
SELECT DBID, PSID, TYPE FROM SYSIBM.SYSTABLESPACE
 WHERE NAME LIKE 'EQ035%'        -- 'TABLESPACE_NAME'
   AND DBNAME='EQ0BP';
SELECT NAME, CREATOR, NAME, OBID FROM SYSIBM.SYSTABLES
 WHERE TSNAME LIKE 'EQ035%'      -- 'TABLESPACE_NAME'
   AND CREATOR='PROD'
;
SELECT DBID, PSID, TYPE FROM SYSIBM.SYSTABLESPACE
WHERE NAME LIKE 'EQ035%'        -- 'TABLESPACE_NAME'
AND DBNAME='AISERIES';
SELECT NAME, CREATOR, NAME, OBID FROM SYSIBM.SYSTABLES
WHERE TSNAME LIKE 'EQ035%'      -- 'TABLESPACE_NAME'
AND CREATOR='SUPPROD'
;
FIM
-- TABLESPACES
 SELECT TS.DBNAME, TS.NAME, TB.NAME, TS.DBID, TS.PSID, TB.OBID
   FROM SYSIBM.SYSTABLESPACE TS
      , SYSIBM.SYSTABLES TB
  WHERE TS.NAME LIKE 'EQ003%'
    AND TS.NAME = TB.TSNAME
 -- AND TB.CREATOR = 'CREATOR_NAME'
    AND SUBSTR(TS.DBNAME, 4, 2) IN ('BP', 'BD')
;
 SELECT TS.DBNAME, TS.NAME, TB.NAME, TS.DBID, TS.PSID, TB.OBID
   FROM SYSIBM.SYSTABLESPACE TS
      , SYSIBM.SYSTABLES TB
  WHERE TS.NAME LIKE 'EQ003%'
    AND TS.NAME = TB.TSNAME
 -- AND TB.CREATOR = 'CREATOR_NAME'
    AND TS.DBNAME = 'AISERIES'
;
